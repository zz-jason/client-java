407    * {@inheritDoc}

409    * @since 1.3.0
410    */
411   @Override
412   public NettyChannelBuilder keepAliveTime(long keepAliveTime, TimeUnit timeUnit) {
413     checkArgument(keepAliveTime > 0L, "keepalive time must be positive");
414     keepAliveTimeNanos = timeUnit.toNanos(keepAliveTime);
415     keepAliveTimeNanos = KeepAliveManager.clampKeepAliveTimeInNanos(keepAliveTimeNanos);
416     if (keepAliveTimeNanos >= AS_LARGE_AS_INFINITE) {
417       // Bump keepalive time to infinite. This disables keepalive.
418       keepAliveTimeNanos = KEEPALIVE_TIME_NANOS_DISABLED;
419     }
420     return this;
421   }

592   private final class DefaultProtocolNegotiator implements ProtocolNegotiator.ClientFactory {
593     private NegotiationType negotiationType = NegotiationType.TLS;
594     private SslContext sslContext;
595 
596     @Override
597     public ProtocolNegotiator newNegotiator() {
598       SslContext localSslContext = sslContext;
599       if (negotiationType == NegotiationType.TLS && localSslContext == null) {
600         try {
601           localSslContext = GrpcSslContexts.forClient().build();
602         } catch (SSLException ex) {
603           throw new RuntimeException(ex);
604         }
605       }
606       return createProtocolNegotiatorByType(
607           negotiationType, localSslContext, managedChannelImplBuilder.getOffloadExecutorPool());
608     }
609 
610     @Override
611     public int getDefaultPort() {
612       switch (negotiationType) {
613         case PLAINTEXT:
614         case PLAINTEXT_UPGRADE:
615           return GrpcUtil.DEFAULT_PORT_PLAINTEXT;
616         case TLS:
617           return GrpcUtil.DEFAULT_PORT_SSL;
618         default:
619           throw new AssertionError(negotiationType + " not handled");
620       }
621     }
622   }

649         Map<ChannelOption<?>, ?> channelOptions,
650         ObjectPool<? extends EventLoopGroup> groupPool,

652         int flowControlWindow,

657         boolean keepAliveWithoutCalls,

659         LocalSocketPicker localSocketPicker,

662       this.channelFactory = channelFactory;

669       this.maxHeaderListSize = maxHeaderListSize;
670       this.keepAliveTimeNanos = keepAliveTimeNanos;
671       this.keepAliveBackoff = new AtomicBackoff("keepalive time nanos", keepAliveTimeNanos);
672       this.keepAliveTimeoutNanos = keepAliveTimeoutNanos;
673       this.keepAliveWithoutCalls = keepAliveWithoutCalls;
674       this.transportTracerFactory = transportTracerFactory;
675       this.localSocketPicker =
676           localSocketPicker != null ? localSocketPicker : new LocalSocketPicker();
677       this.useGetForSafeMethods = useGetForSafeMethods;

680     @Override
681     public ConnectionClientTransport newClientTransport(
682         SocketAddress serverAddress, ClientTransportOptions options, ChannelLogger channelLogger) {
683       checkState(!closed, "The transport factory is closed.");
684 
685       ProtocolNegotiator localNegotiator = protocolNegotiator;
686       HttpConnectProxiedSocketAddress proxiedAddr = options.getHttpConnectProxiedSocketAddress();
687       if (proxiedAddr != null) {
688         serverAddress = proxiedAddr.getTargetAddress();
689         localNegotiator =
690             ProtocolNegotiators.httpProxy(
691                 proxiedAddr.getProxyAddress(),
692                 proxiedAddr.getUsername(),
693                 proxiedAddr.getPassword(),
694                 protocolNegotiator);
695       }
696 
697       final AtomicBackoff.State keepAliveTimeNanosState = keepAliveBackoff.getState();
698       Runnable tooManyPingsRunnable =
699           new Runnable() {
700             @Override
701             public void run() {
702               keepAliveTimeNanosState.backoff();
703             }
704           };
705 
706       // TODO(carl-mastrangelo): Pass channelLogger in.
707       NettyClientTransport transport =
708           new NettyClientTransport(
709               serverAddress,
710               channelFactory,
711               channelOptions,
712               group,
713               localNegotiator,
714               autoFlowControl,
715               flowControlWindow,
716               maxMessageSize,
717               maxHeaderListSize,
718               keepAliveTimeNanosState.get(),
719               keepAliveTimeoutNanos,
720               keepAliveWithoutCalls,
721               options.getAuthority(),
722               options.getUserAgent(),
723               tooManyPingsRunnable,
724               transportTracerFactory.create(),
725               options.getEagAttributes(),
726               localSocketPicker,
727               channelLogger,
728               useGetForSafeMethods);
729       return transport;
730     }

737     @Override
738     public SwapChannelCredentialsResult swapChannelCredentials(ChannelCredentials channelCreds) {
739       checkNotNull(channelCreds, "channelCreds");
740       FromChannelCredentialsResult result = ProtocolNegotiators.from(channelCreds);
741       if (result.error != null) {
742         return null;
743       }
744       ClientTransportFactory factory =
745           new NettyTransportFactory(
746               result.negotiator.newNegotiator(),
747               channelFactory,
748               channelOptions,
749               groupPool,
750               autoFlowControl,
751               flowControlWindow,
752               maxMessageSize,
753               maxHeaderListSize,
754               keepAliveTimeNanos,
755               keepAliveTimeoutNanos,
756               keepAliveWithoutCalls,
757               transportTracerFactory,
758               localSocketPicker,
759               useGetForSafeMethods);
760       return new SwapChannelCredentialsResult(factory, result.callCredentials);
761     }

765       if (closed) {
766         return;
767       }

771       groupPool.returnObject(group);
