34  /** A queue of pending writes to a {@link Channel} that is flushed as a single unit. */
35  class WriteQueue {
36  
37    // Dequeue in chunks, so we don't have to acquire the queue's log too often.
38    @VisibleForTesting static final int DEQUE_CHUNK_SIZE = 128;
39  
40    /** {@link Runnable} used to schedule work onto the tail of the event loop. */
41    private final Runnable later =
42        new Runnable() {
43          @Override
44          public void run() {
45            flush();
46          }
47        };

49    private final Channel channel;
50    private final Queue<Pair<QueuedCommand, Long>> queue;
51    private final AtomicBoolean scheduled = new AtomicBoolean();

100   public WriteQueue(Channel channel) {
101     this.channel = Preconditions.checkNotNull(channel, "channel");
102     queue = new ConcurrentLinkedQueue<>();
103   }
104 
105   /** Schedule a flush on the channel. */
106   void scheduleFlush() {
107     if (scheduled.compareAndSet(false, true)) {
108       // Add the queue to the tail of the event loop so writes will be executed immediately
109       // inside the event loop. Note DO NOT do channel.write outside the event loop as
110       // it will not wake up immediately without a flush.
111       channel.eventLoop().execute(later);
112     }
113   }
114 
115   /**
116    * Enqueue a write command on the channel.
117    *
118    * @param command a write to be executed on the channel.
119    * @param flush true if a flush of the write should be schedule, false if a later call to enqueue
120    *     will schedule the flush.
121    */
122   @CanIgnoreReturnValue
123   ChannelFuture enqueue(QueuedCommand command, boolean flush) {
124     // Detect erroneous code that tries to reuse command objects.
125     Preconditions.checkArgument(command.promise() == null, "promise must not be set on command");
126 
127     ChannelPromise promise = channel.newPromise();
128     command.promise(promise);
129     queue.add(Pair.of(command, System.nanoTime()));
130     if (flush) {
131       scheduleFlush();
132     }
133     return promise;
134   }
135 
136   /**
137    * Enqueue the runnable. It is not safe for another thread to queue an Runnable directly to the
138    * event loop, because it will be out-of-order with writes. This method allows the Runnable to be
139    * processed in-order with writes.
140    */
141   void enqueue(Runnable runnable, boolean flush) {
142     Long now = System.nanoTime();
143     queue.add(Pair.<QueuedCommand, Long>of(new RunnableCommand(runnable), now));
144     if (flush) {
145       scheduleFlush();
146     }
147   }
148 
149   /**
150    * Executes enqueued work directly on the current thread. This can be used to trigger writes
151    * before performing additional reads. Must be called from the event loop. This method makes no
152    * guarantee that the work queue is empty when it returns.
153    */
154   void drainNow() {
155     Preconditions.checkState(channel.eventLoop().inEventLoop(), "must be on the event loop");
156     if (queue.peek() == null) {
157       return;
158     }
159     flush();
160   }
161 
162   /**
163    * Process the queue of commands and dispatch them to the stream. This method is only called in
164    * the event loop
165    */
166   private void flush() {
167     Histogram.Timer flushTimer = writeQueueFlushDuration.startTimer();
168     PerfMark.startTask("WriteQueue.periodicFlush");
169     Histogram.Timer periodicFlush =
170         perfmarkWriteQueueDuration.labels("WriteQueue.periodicFlush").startTimer();
171 
172     long start = System.nanoTime();
173     try {
174       Pair<QueuedCommand, Long> item;
175       int i = 0;
176       boolean flushedOnce = false;
177       Histogram.Timer waitBatchTimer = writeQueueWaitBatchDuration.startTimer();
178       while ((item = queue.poll()) != null) {
179         QueuedCommand cmd = item.getLeft();
180         String cmdName = cmd.getClass().getSimpleName();
181         writeQueuePendingDuration
182             .labels(cmdName)
183             .observe((System.nanoTime() - item.getRight()) / 1_000_000.0);
184 
185         Histogram.Timer cmdTimer = writeQueueCmdRunDuration.labels(cmdName).startTimer();
186 
187         // Run the command
188         cmd.run(channel);
189 
190         cmdTimer.observeDuration();
191 
192         if (++i == DEQUE_CHUNK_SIZE) {
193           waitBatchTimer.observeDuration();
194           i = 0;
195           // Flush each chunk so we are releasing buffers periodically. In theory this loop
196           // might never end as new events are continuously added to the queue, if we never
197           // flushed in that case we would be guaranteed to OOM.
198           PerfMark.startTask("WriteQueue.flush0");
199           Histogram.Timer flush0 =
200               perfmarkWriteQueueDuration.labels("WriteQueue.flush0").startTimer();
201           Histogram.Timer channelFlushTimer =
202               writeQueueChannelFlushDuration.labels("flush0").startTimer();
203           try {
204             channel.flush();
205           } finally {
206             waitBatchTimer = writeQueueWaitBatchDuration.startTimer();
207             writeQueueBatchSize.observe(DEQUE_CHUNK_SIZE);
208             channelFlushTimer.observeDuration();
209             PerfMark.stopTask("WriteQueue.flush0");
210             flush0.observeDuration();
211           }
212           flushedOnce = true;
213         }
214       }
215       // Must flush at least once, even if there were no writes.
216       if (i != 0 || !flushedOnce) {
217         waitBatchTimer.observeDuration();
218         PerfMark.startTask("WriteQueue.flush1");
219         Histogram.Timer flush1 =
220             perfmarkWriteQueueDuration.labels("WriteQueue.flush1").startTimer();
221         Histogram.Timer channelFlushTimer =
222             writeQueueChannelFlushDuration.labels("flush1").startTimer();
223         try {
224           channel.flush();
225         } finally {
226           writeQueueBatchSize.observe(i);
227           channelFlushTimer.observeDuration();
228           PerfMark.stopTask("WriteQueue.flush1");
229           flush1.observeDuration();
230         }
231       }
232     } finally {
233       PerfMark.stopTask("WriteQueue.periodicFlush");
234       periodicFlush.observeDuration();
235       flushTimer.observeDuration();
236       // Mark the write as done, if the queue is non-empty after marking trigger a new write.
237       scheduled.set(false);
238       if (!queue.isEmpty()) {
239         scheduleFlush();
240       }
241     }
242   }

244   private static class RunnableCommand implements QueuedCommand {
245     private final Runnable runnable;
246     private final Link link;
247 
248     public RunnableCommand(Runnable runnable) {
249       this.link = PerfMark.linkOut();
250       this.runnable = runnable;
251     }
252 
253     @Override
254     public final void promise(ChannelPromise promise) {
255       throw new UnsupportedOperationException();
256     }
257 
258     @Override
259     public final ChannelPromise promise() {
260       throw new UnsupportedOperationException();
261     }
262 
263     @Override
264     public final void run(Channel channel) {
265       runnable.run();
266     }
267 
268     @Override
269     public Link getLink() {
270       return link;
271     }
272   }

274   abstract static class AbstractQueuedCommand implements QueuedCommand {
275 
276     private ChannelPromise promise;
277     private final Link link;
278 
279     AbstractQueuedCommand() {
280       this.link = PerfMark.linkOut();
281     }
282 
283     @Override
284     public final void promise(ChannelPromise promise) {
285       this.promise = promise;
286     }
287 
288     @Override
289     public final ChannelPromise promise() {
290       return promise;
291     }
292 
293     @Override
294     public final void run(Channel channel) {
295       channel.write(this, promise);
296     }
297 
298     @Override
299     public Link getLink() {
300       return link;
301     }
302   }

304   /** Simple wrapper type around a command and its optional completion listener. */
305   interface QueuedCommand {
306     /** Returns the promise beeing notified of the success/failure of the write. */
307     ChannelPromise promise();
308 
309     /** Sets the promise. */
310     void promise(ChannelPromise promise);
311 
312     void run(Channel channel);
313 
314     Link getLink();
315   }
