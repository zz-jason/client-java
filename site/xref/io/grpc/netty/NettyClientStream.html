52  /** Client stream for a Netty transport. Must only be called from the sending application thread. */
53  class NettyClientStream extends AbstractClientStream {
54    private static final InternalMethodDescriptor methodDescriptorAccessor =
55        new InternalMethodDescriptor(
56            NettyClientTransport.class.getName().contains("grpc.netty.shaded")
57                ? InternalKnownTransport.NETTY_SHADED
58                : InternalKnownTransport.NETTY);

69        HistogramUtils.buildDuration()
70            .name("perfmark_netty_client_stream_duration_seconds")
71            .help("Perfmark netty client stream duration seconds")
72            .labelNames("type")
73            .register();

75    NettyClientStream(
76        TransportState state,
77        MethodDescriptor<?, ?> method,
78        Metadata headers,
79        Channel channel,
80        AsciiString authority,
81        AsciiString scheme,
82        AsciiString userAgent,
83        StatsTraceContext statsTraceCtx,
84        TransportTracer transportTracer,
85        CallOptions callOptions,
86        boolean useGetForSafeMethods) {
87      super(
88          new NettyWritableBufferAllocator(channel.alloc()),
89          statsTraceCtx,
90          transportTracer,
91          headers,
92          callOptions,
93          useGetForSafeMethods && method.isSafe());
94      this.state = checkNotNull(state, "transportState");
95      this.writeQueue = state.handler.getWriteQueue();
96      this.method = checkNotNull(method, "method");
97      this.authority = checkNotNull(authority, "authority");
98      this.scheme = checkNotNull(scheme, "scheme");
99      this.userAgent = userAgent;
100   }

122   private class Sink implements AbstractClientStream.Sink {
123 
124     @Override
125     public void writeHeaders(Metadata headers, byte[] requestPayload) {
126       PerfMark.startTask("NettyClientStream$Sink.writeHeaders");
127       Histogram.Timer writeHeaders =
128           perfmarkNettyClientStreamDuration
129               .labels("NettyClientStream$Sink.writeHeaders")
130               .startTimer();
131       try {
132         writeHeadersInternal(headers, requestPayload);
133       } finally {
134         PerfMark.stopTask("NettyClientStream$Sink.writeHeaders");
135         writeHeaders.observeDuration();
136       }
137     }
138 
139     private void writeHeadersInternal(Metadata headers, byte[] requestPayload) {
140       // Convert the headers into Netty HTTP/2 headers.
141       AsciiString defaultPath = (AsciiString) methodDescriptorAccessor.geRawMethodName(method);
142       if (defaultPath == null) {
143         defaultPath = new AsciiString("/" + method.getFullMethodName());
144         methodDescriptorAccessor.setRawMethodName(method, defaultPath);
145       }
146       boolean get = (requestPayload != null);
147       AsciiString httpMethod;
148       if (get) {
149         // Forge the query string
150         // TODO(ericgribkoff) Add the key back to the query string
151         defaultPath =
152             new AsciiString(defaultPath + "?" + BaseEncoding.base64().encode(requestPayload));
153         httpMethod = Utils.HTTP_GET_METHOD;
154       } else {
155         httpMethod = Utils.HTTP_METHOD;
156       }
157       Http2Headers http2Headers =
158           Utils.convertClientHeaders(
159               headers, scheme, defaultPath, authority, httpMethod, userAgent);
160 
161       ChannelFutureListener failureListener =
162           new ChannelFutureListener() {
163             @Override
164             public void operationComplete(ChannelFuture future) throws Exception {
165               if (!future.isSuccess()) {
166                 // Stream creation failed. Close the stream if not already closed.
167                 // When the channel is shutdown, the lifecycle manager has a better view of the
168                 // failure,
169                 // especially before negotiation completes (because the negotiator commonly doesn't
170                 // receive the execeptionCaught because NettyClientHandler does not propagate it).
171                 Status s = transportState().handler.getLifecycleManager().getShutdownStatus();
172                 if (s == null) {
173                   s = transportState().statusFromFailedFuture(future);
174                 }
175                 transportState().transportReportStatus(s, true, new Metadata());
176               }
177             }
178           };
179       // Write the command requesting the creation of the stream.
180       writeQueue
181           .enqueue(
182               new CreateStreamCommand(
183                   http2Headers, transportState(), shouldBeCountedForInUse(), get),
184               !method.getType().clientSendsOneMessage() || get)
185           .addListener(failureListener);
186     }
187 
188     private void writeFrameInternal(
189         WritableBuffer frame, boolean endOfStream, boolean flush, final int numMessages) {
190       Preconditions.checkArgument(numMessages >= 0);
191       ByteBuf bytebuf =
192           frame == null ? EMPTY_BUFFER : ((NettyWritableBuffer) frame).bytebuf().touch();
193       final int numBytes = bytebuf.readableBytes();
194       if (numBytes > 0) {
195         // Add the bytes to outbound flow control.
196         onSendingBytes(numBytes);
197         writeQueue
198             .enqueue(new SendGrpcFrameCommand(transportState(), bytebuf, endOfStream), flush)
199             .addListener(
200                 new ChannelFutureListener() {
201                   @Override
202                   public void operationComplete(ChannelFuture future) throws Exception {
203                     // If the future succeeds when http2stream is null, the stream has been
204                     // cancelled
205                     // before it began and Netty is purging pending writes from the flow-controller.
206                     if (future.isSuccess() && transportState().http2Stream() != null) {
207                       // Remove the bytes from outbound flow control, optionally notifying
208                       // the client that they can send more bytes.
209                       transportState().onSentBytes(numBytes);
210                       NettyClientStream.this.getTransportTracer().reportMessageSent(numMessages);
211                     }
212                   }
213                 });
214       } else {
215         // The frame is empty and will not impact outbound flow control. Just send it.
216         writeQueue.enqueue(new SendGrpcFrameCommand(transportState(), bytebuf, endOfStream), flush);
217       }
218     }
219 
220     @Override
221     public void writeFrame(
222         WritableBuffer frame, boolean endOfStream, boolean flush, int numMessages) {
223       PerfMark.startTask("NettyClientStream$Sink.writeFrame");
224       Histogram.Timer writeFrame =
225           perfmarkNettyClientStreamDuration
226               .labels("NettyClientStream$Sink.writeFrame")
227               .startTimer();
228       try {
229         writeFrameInternal(frame, endOfStream, flush, numMessages);
230       } finally {
231         PerfMark.stopTask("NettyClientStream$Sink.writeFrame");
232         writeFrame.observeDuration();
233       }
234     }
235 
236     @Override
237     public void cancel(Status status) {
238       PerfMark.startTask("NettyClientStream$Sink.cancel");
239       Histogram.Timer cancel =
240           perfmarkNettyClientStreamDuration.labels("NettyClientStream$Sink.cancel").startTimer();
241       try {
242         writeQueue.enqueue(new CancelClientStreamCommand(transportState(), status), true);
243       } finally {
244         PerfMark.stopTask("NettyClientStream$Sink.cancel");
245         cancel.observeDuration();
246       }
247     }

250   /** This should only called from the transport thread. */
251   public abstract static class TransportState extends Http2ClientStreamTransportState
252       implements StreamIdHolder {
253     private static final int NON_EXISTENT_ID = -1;
254 
255     private final String methodName;
256     private final NettyClientHandler handler;
257     private final EventLoop eventLoop;
258     private int id;
259     private Http2Stream http2Stream;
260     private Tag tag;
261 
262     protected TransportState(
263         NettyClientHandler handler,
264         EventLoop eventLoop,
265         int maxMessageSize,
266         StatsTraceContext statsTraceCtx,
267         TransportTracer transportTracer,
268         String methodName) {
269       super(maxMessageSize, statsTraceCtx, transportTracer);
270       this.methodName = checkNotNull(methodName, "methodName");
271       this.handler = checkNotNull(handler, "handler");
272       this.eventLoop = checkNotNull(eventLoop, "eventLoop");
273       tag = PerfMark.createTag(methodName);
274     }
275 
276     @Override
277     public int id() {
278       // id should be positive
279       return id;
280     }
281 
282     public void setId(int id) {
283       checkArgument(id > 0, "id must be positive %s", id);
284       checkState(this.id == 0, "id has been previously set: %s", this.id);
285       this.id = id;
286       this.tag = PerfMark.createTag(methodName, id);
287     }
288 
289     /**
290      * Marks the stream state as if it had never existed. This can happen if the stream is cancelled
291      * after it is created, but before it has been started.
292      */
293     void setNonExistent() {
294       checkState(this.id == 0, "Id has been previously set: %s", this.id);
295       this.id = NON_EXISTENT_ID;
296     }
297 
298     boolean isNonExistent() {
299       return this.id == NON_EXISTENT_ID;
300     }
301 
302     /**
303      * Sets the underlying Netty {@link Http2Stream} for this stream. This must be called in the
304      * context of the transport thread.
305      */
306     public void setHttp2Stream(Http2Stream http2Stream) {
307       checkNotNull(http2Stream, "http2Stream");
308       checkState(this.http2Stream == null, "Can only set http2Stream once");
309       this.http2Stream = http2Stream;
310 
311       // Now that the stream has actually been initialized, call the listener's onReady callback if
312       // appropriate.
313       onStreamAllocated();
314       getTransportTracer().reportLocalStreamStarted();
315     }
316 
317     /** Gets the underlying Netty {@link Http2Stream} for this stream. */
318     @Nullable
319     public Http2Stream http2Stream() {
320       return http2Stream;
321     }
322 
323     /**
324      * Intended to be overridden by NettyClientTransport, which has more information about failures.
325      * May only be called from event loop.
326      */
327     protected abstract Status statusFromFailedFuture(ChannelFuture f);
328 
329     @Override
330     protected void http2ProcessingFailed(Status status, boolean stopDelivery, Metadata trailers) {
331       transportReportStatus(status, stopDelivery, trailers);
332       handler.getWriteQueue().enqueue(new CancelClientStreamCommand(this, status), true);
333     }
334 
335     @Override
336     public void runOnTransportThread(final Runnable r) {
337       if (eventLoop.inEventLoop()) {
338         r.run();
339       } else {
340         eventLoop.execute(r);
341       }
342     }
343 
344     @Override
345     public void bytesRead(int processedBytes) {
346       handler.returnProcessedBytes(http2Stream, processedBytes);
347       handler.getWriteQueue().scheduleFlush();
348     }
349 
350     @Override
351     public void deframeFailed(Throwable cause) {
352       http2ProcessingFailed(Status.fromThrowable(cause), true, new Metadata());
353     }
354 
355     void transportHeadersReceived(Http2Headers headers, boolean endOfStream) {
356       if (endOfStream) {
357         if (!isOutboundClosed()) {
358           handler.getWriteQueue().enqueue(new CancelClientStreamCommand(this, null), true);
359         }
360         transportTrailersReceived(Utils.convertTrailers(headers));
361       } else {
362         transportHeadersReceived(Utils.convertHeaders(headers));
363       }
364     }
365 
366     void transportDataReceived(ByteBuf frame, boolean endOfStream) {
367       transportDataReceived(new NettyReadableBuffer(frame.retain()), endOfStream);
368     }
369 
370     @Override
371     public final Tag tag() {
372       return tag;
373     }
374   }
